# 1. Make sure you're on main
git switch main

# 2. Make main track the PRETACE repository (origin), not the old TACE repository
git branch --set-upstream-to=origin/main main

# 3. Remove the old remote (optional, but recommended)
git remote remove tace-old

# 4. Ignore temporary FUSE files forever
echo ".fuse_hidden*" >> .gitignore

# 5. Remove any FUSE temporary files from Git
git rm --cached .fuse_hidden*

# 6. Remove them from disk as well (optional)
rm -f .fuse_hidden*

# 7. Commit these cleanup changes
git add .gitignore
git commit -m "Configure main for pretace and ignore FUSE temporary files"

# 8. Push to PRETACE
git push

# 9. Verify everything
git remote -v
git branch -vv
